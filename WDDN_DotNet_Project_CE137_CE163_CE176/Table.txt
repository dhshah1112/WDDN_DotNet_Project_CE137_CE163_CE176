CREATE TABLE [dbo].[userDetails]
(
	[Id] INT NOT NULL PRIMARY KEY IDENTITY, 
    [userName] NVARCHAR(50) NOT NULL, 
    [Contact] NVARCHAR(50) NOT NULL, 
    [Email] NVARCHAR(50) NOT NULL, 
    [Name] NVARCHAR(50) NOT NULL, 
    [Password] NVARCHAR(50) NOT NULL,

)


CREATE TABLE [dbo].[Admin]
(
	[Id] INT NOT NULL PRIMARY KEY IDENTITY, 
    [userName] NVARCHAR(50) NOT NULL, 
    [Password] NVARCHAR(50) NOT NULL, 
    [Email] NVARCHAR(50) NOT NULL
)

CREATE TABLE [dbo].[Ticket] (
    [Id]            INT           IDENTITY (1, 1) NOT NULL,
    [MovieName]     NVARCHAR (50) NOT NULL,
    [SilverPrice]   NVARCHAR (10) NOT NULL,
    [GoldPrice]     NVARCHAR (10) NOT NULL,
    [PlatinumPrice] NVARCHAR (10) NOT NULL,
    [Description]   NCHAR (1000)  NOT NULL,
    [Image]         NVARCHAR(MAX)  NOT NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);


CREATE TABLE [dbo].[Bookings]
(
	[Id] INT NOT NULL PRIMARY KEY, 
    [userId] INT NOT NULL
	CONSTRAINT userId_fk FOREIGN KEY (userId) REFERENCES userDetails(Id),
    [Total] INT NOT NULL, 
    [ticketId] INT NOT NULL 
	CONSTRAINT ticketId_fk FOREIGN KEY (ticketId) REFERENCES Ticket(Id), 
    [showTime] NVARCHAR(50) NOT NULL 
)
